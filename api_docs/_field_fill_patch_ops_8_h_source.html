383  void set_inflow(
384  int lev,
385  amrex::Real time,
386  amrex::MultiFab& mfab,
387  const amrex::IntVect& nghost,
388  const FieldState /*fstate*/) override
389  {
390  const int ng = nghost[0];
391  const auto& bctype = m_field.bc_type();
392  const auto& geom = m_mesh.Geom(lev);
393  const auto& gdata = geom.data();
394  const auto& domain = geom.growPeriodicDomain(ng);
395  const auto& bcfunc = bc_functor();
396  const auto& ncomp = m_field.num_comp();
397 
398  for (amrex::OrientationIter oit; oit != nullptr; ++oit) {
399  auto ori = oit();
400  if (bctype[ori] != BC::mass_inflow) {
401  continue;
402  }
403 
404  const int idir = ori.coordDir();
405  const auto& dbx =
406  ori.isLow() ? amrex::adjCellLo(domain, idir, nghost[idir])
407  : amrex::adjCellHi(domain, idir, nghost[idir]);
408 
409  for (amrex::MFIter mfi(mfab); mfi.isValid(); ++mfi) {
410  const auto& gbx = amrex::grow(mfi.validbox(), nghost);
411  const auto& bx = gbx & dbx;
412  if (!bx.ok()) {
413  continue;
414  }
415 
416  const auto& marr = mfab[mfi].array();
417  amrex::ParallelFor(
418  bx, [=] AMREX_GPU_DEVICE(
419  const int i, const int j, const int k) noexcept {
420  for (int n = 0; n < ncomp; ++n) {
421  bcfunc.set_inflow(
422  {i, j, k}, marr, gdata, time, ori, n, 0, 0);
423  }
424  });
425  }
426  }
427  }
