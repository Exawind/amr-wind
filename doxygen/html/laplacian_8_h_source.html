23 {
24 const int ncomp = m_phi.num_comp();
25 const auto& geom = m_phi.repo().mesh().Geom(lev);
26 const auto& idx = geom.InvCellSizeArray();
27 const auto& lapphi = m_lapphi(lev).array(mfi);
28 const auto& phi = m_phi(lev).const_array(mfi);
29
30 const auto& bx_in = mfi.tilebox();
31 const auto& bx = Stencil::box(bx_in, geom);
32 if (bx.isEmpty()) {
33 return;
34 }
35
36 amrex::ParallelFor(bx, [=] AMREX_GPU_DEVICE(int i, int j, int k) {
37 for (int icomp = 0; icomp < ncomp; icomp++) {
38 amrex::Real sp1 = Stencil::s00;
39 amrex::Real s = Stencil::s01;
40 amrex::Real sm1 = Stencil::s02;
41 amrex::Real d2phidx2 =
42 (sp1 * phi(i + 1, j, k, 0) + s * phi(i, j, k, 0) +
43 sm1 * phi(i - 1, j, k, 0)) *
44 idx[0] * idx[0];
45 sp1 = Stencil::s10;
46 s = Stencil::s11;
47 sm1 = Stencil::s12;
48 amrex::Real d2phidy2 =
49 (sp1 * phi(i, j + 1, k, 1) + s * phi(i, j, k, 1) +
50 sm1 * phi(i, j - 1, k, 1)) *
51 idx[1] * idx[1];
52 sp1 = Stencil::s20;
53 s = Stencil::s21;
54 sm1 = Stencil::s22;
55 amrex::Real d2phidz2 =
56 (sp1 * phi(i, j, k + 1, 2) + s * phi(i, j, k, 2) +
57 sm1 * phi(i, j, k - 1, 2)) *
58 idx[2] * idx[2];
59 lapphi(i, j, k) = d2phidx2 + d2phidy2 + d2phidz2;
60 }
61 });
