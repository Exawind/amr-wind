32 AdvectionOp(
33 CFDSim& /* sim */,
34 PDEFields& fields_in,
35 bool /* has_overset */,
36 bool /* variable density */,
37 bool /* mesh mapping */,
38 bool /* is_anelastic */)
39 : fields(fields_in)
40 , density(fields_in.repo.get_field("density"))
41 , u_mac(fields_in.repo.get_field("u_mac"))
42 , v_mac(fields_in.repo.get_field("v_mac"))
43 , w_mac(fields_in.repo.get_field("w_mac"))
44 {
45 amrex::ParmParse pp("incflo");
46 pp.query("godunov_type", godunov_type);
47 pp.query("godunov_use_forces_in_trans", godunov_use_forces_in_trans);
48 if (pp.contains("use_ppm") || pp.contains("use_limiter")) {
49 amrex::Abort(
50 "Godunov: use_ppm and use_limiter are deprecated. Please "
51 "update input file");
52 }
53
54 if (amrex::toLower(godunov_type) == "plm") {
55 godunov_scheme = godunov::scheme::PLM;
56 } else if (amrex::toLower(godunov_type) == "ppm") {
57 godunov_scheme = godunov::scheme::PPM;
58 } else if (amrex::toLower(godunov_type) == "ppm_nolim") {
59 godunov_scheme = godunov::scheme::PPM_NOLIM;
60 amrex::Print() << "WARNING: Using advection type ppm_nolim is not "
61 "recommended. Prefer using weno_z."
62 << '\n';
63 } else if (amrex::toLower(godunov_type) == "bds") {
64 godunov_scheme = godunov::scheme::BDS;
65 advection_type = "BDS";
66 } else if (
67 amrex::toLower(godunov_type) == "weno" ||
68 amrex::toLower(godunov_type) == "weno_js") {
69 godunov_scheme = godunov::scheme::WENO_JS;
70 } else if (amrex::toLower(godunov_type) == "weno_z") {
71 godunov_scheme = godunov::scheme::WENOZ;
72 } else {
73 amrex::Abort(
74 "Invalid godunov_type specified. For godunov_type select "
75 "between plm, ppm, ppm_nolim, bds, weno_js, and weno_z. If no "
76 "godunov_type is specified, the default weno_z is used.");
77 }
78
79 // Flux calculation used in multiphase portions of domain
80 pp.query("mflux_type", mflux_type);
81 if (amrex::toLower(mflux_type) == "minmod") {
82 mflux_scheme = godunov::scheme::MINMOD;
83 } else if (amrex::toLower(mflux_type) == "upwind") {
84 mflux_scheme = godunov::scheme::UPWIND;
85 } else {
86 amrex::Abort("Invalid argument entered for mflux_type.");
87 }
88
89 amrex::ParmParse pp_eq{fields_in.field.base_name()};
90 pp_eq.query(
91 "allow_inflow_at_pressure_outflow", m_allow_inflow_on_outflow);
92
93 // TODO: Need iconserv flag to be adjusted???
94 iconserv.resize(PDE::ndim, 1);
95 }

103 void operator()(const FieldState fstate, const amrex::Real dt)
104 {
105 static_assert(
106 PDE::ndim == 1, "Invalid number of components for scalar");
107 auto& repo = fields.repo;
108 const auto& geom = repo.mesh().Geom();
109
110 const auto& src_term = fields.src_term;
111 // cppcheck-suppress constVariableReference
112 auto& conv_term = fields.conv_term;
113 const auto& dof_field = fields.field.state(fstate);
114 const auto& dof_nph = fields.field.state(amr_wind::FieldState::NPH);
115
116 auto flux_x =
117 repo.create_scratch_field(PDE::ndim, 0, amr_wind::FieldLoc::XFACE);
118 auto flux_y =
119 repo.create_scratch_field(PDE::ndim, 0, amr_wind::FieldLoc::YFACE);
120 auto flux_z =
121 repo.create_scratch_field(PDE::ndim, 0, amr_wind::FieldLoc::ZFACE);
122 auto face_x =
123 repo.create_scratch_field(PDE::ndim, 0, amr_wind::FieldLoc::XFACE);
124 auto face_y =
125 repo.create_scratch_field(PDE::ndim, 0, amr_wind::FieldLoc::YFACE);
126 auto face_z =
127 repo.create_scratch_field(PDE::ndim, 0, amr_wind::FieldLoc::ZFACE);
128
129 // only needed if multiplying by rho below
130 const auto& den = density.state(fstate);
131 const auto& den_nph = density.state(amr_wind::FieldState::NPH);
132
133 const bool mphase_vof = repo.field_exists("vof");
134
135 for (int lev = 0; lev < repo.num_active_levels(); ++lev) {
136 amrex::MFItInfo mfi_info;
137 if (amrex::Gpu::notInLaunchRegion()) {
138 mfi_info.EnableTiling(amrex::IntVect(1024, 1024, 1024))
139 .SetDynamic(true);
140 }
141#ifdef AMREX_USE_OMP
142#pragma omp parallel if (amrex::Gpu::notInLaunchRegion())
143#endif
144 for (amrex::MFIter mfi(dof_field(lev), mfi_info); mfi.isValid();
145 ++mfi) {
146 const auto& bx = mfi.tilebox();
147 auto rho_arr = den(lev).array(mfi);
148 auto rho_nph_arr = den_nph(lev).array(mfi);
149 auto tra_arr = dof_field(lev).array(mfi);
150 auto tra_nph_arr = dof_nph(lev).array(mfi);
151 amrex::FArrayBox rhotracfab;
152 amrex::Array4<amrex::Real> rhotrac;
153 amrex::FArrayBox rhotrac_nph_fab;
154 amrex::Array4<amrex::Real> rhotrac_nph;
155 amrex::FArrayBox srctrac_over_rho_fab;
156 amrex::Array4<amrex::Real> srctrac_over_rho;
157 const auto& src_arr = src_term(lev).const_array(mfi);
158
159 if (PDE::multiply_rho) {
160 auto rhotrac_box =
161 amrex::grow(bx, fvm::Godunov::nghost_state);
162 auto srctrac_box =
163 amrex::grow(bx, fvm::Godunov::nghost_src);
164 rhotracfab.resize(
165 rhotrac_box, PDE::ndim, amrex::The_Async_Arena());
166 rhotrac = rhotracfab.array();
167 rhotrac_nph_fab.resize(
168 rhotrac_box, PDE::ndim, amrex::The_Async_Arena());
169 rhotrac_nph = rhotrac_nph_fab.array();
170 if (mphase_vof) {
171 srctrac_over_rho_fab.resize(
172 rhotrac_box, PDE::ndim, amrex::The_Async_Arena());
173 srctrac_over_rho = srctrac_over_rho_fab.array();
174 }
175
176 amrex::ParallelFor(
177 rhotrac_box, PDE::ndim,
178 [=] AMREX_GPU_DEVICE(int i, int j, int k, int n) {
179 rhotrac(i, j, k, n) =
180 rho_arr(i, j, k) * tra_arr(i, j, k, n);
181 rhotrac_nph(i, j, k, n) =
182 rho_nph_arr(i, j, k) * tra_nph_arr(i, j, k, n);
183 if (mphase_vof && srctrac_box.contains(i, j, k)) {
184 // Need to remove the effect of the multiply_rho
185 // routine -- the setup is different for
186 // mass-consistent multiphase advection
187 srctrac_over_rho(i, j, k, n) =
188 src_arr(i, j, k, n) / rho_nph_arr(i, j, k);
189 }
190 });
191 }
192
193 if ((godunov_scheme == godunov::scheme::PPM) ||
194 (godunov_scheme == godunov::scheme::PPM_NOLIM) ||
195 (godunov_scheme == godunov::scheme::PLM) ||
196 (godunov_scheme == godunov::scheme::WENOZ) ||
197 (godunov_scheme == godunov::scheme::WENO_JS) ||
198 (godunov_scheme == godunov::scheme::BDS)) {
199 amrex::FArrayBox tmpfab(
200 amrex::grow(bx, 1), 1, amrex::The_Async_Arena());
201 tmpfab.setVal<amrex::RunOn::Device>(0.0_rt);
202 const auto& divu = tmpfab.array();
203 const bool is_velocity = false;
204 const bool known_edge_state = false;
205 const bool godunov_use_ppm =
206 ((godunov_scheme != godunov::scheme::PLM) &&
207 (godunov_scheme != godunov::scheme::BDS));
208 int limiter_type;
209 if (godunov_scheme == godunov::scheme::PPM_NOLIM) {
210 limiter_type = PPM::NoLimiter;
211 } else if (godunov_scheme == godunov::scheme::WENOZ) {
212 limiter_type = PPM::WENOZ;
213 } else if (godunov_scheme == godunov::scheme::WENO_JS) {
214 limiter_type = PPM::WENO_JS;
215 } else {
216 limiter_type = PPM::default_limiter;
217 }
218
219 // if state is NPH, then n and n+1 are known, and only
220 // spatial extrapolation is performed
221 const amrex::Real dt_extrap =
222 (fstate == FieldState::NPH) ? 0.0_rt : dt;
223
224 HydroUtils::ComputeFluxesOnBoxFromState(
225 bx, PDE::ndim, mfi,
226 ((PDE::multiply_rho && !mphase_vof) ? rhotrac
227 : tra_arr),
228 ((PDE::multiply_rho && !mphase_vof) ? rhotrac_nph
229 : tra_nph_arr),
230 (*flux_x)(lev).array(mfi), (*flux_y)(lev).array(mfi),
231 (*flux_z)(lev).array(mfi), (*face_x)(lev).array(mfi),
232 (*face_y)(lev).array(mfi), (*face_z)(lev).array(mfi),
233 known_edge_state, u_mac(lev).const_array(mfi),
234 v_mac(lev).const_array(mfi),
235 w_mac(lev).const_array(mfi), divu,
236 ((PDE::multiply_rho && mphase_vof)
237 ? srctrac_over_rho
238 : src_term(lev).const_array(mfi)),
239 geom[lev], dt_extrap, dof_field.bcrec(),
240 dof_field.bcrec_device().data(), iconserv.data(),
241 godunov_use_ppm, godunov_use_forces_in_trans,
242 is_velocity, fluxes_are_area_weighted, advection_type,
243 limiter_type, m_allow_inflow_on_outflow);
244 } else {
245 amrex::Abort("Invalid godunov scheme");
246 }
247 }
248 }
249
250 // Multiphase flux operations
251 if (mphase_vof && PDE::multiply_rho) {
252 // Loop levels
253 multiphase::hybrid_fluxes(
254 repo, PDE::ndim, iconserv, (*flux_x), (*flux_y), (*flux_z),
255 dof_field, dof_nph, src_term, den, den_nph, u_mac, v_mac, w_mac,
256 dof_field.bcrec(), dof_field.bcrec_device().data(), den.bcrec(),
257 den.bcrec_device().data(), dt, mflux_scheme,
258 m_allow_inflow_on_outflow, godunov_use_forces_in_trans, true);
259 }
260
261 amrex::Vector<amrex::Array<amrex::MultiFab*, AMREX_SPACEDIM>> fluxes(
262 repo.num_active_levels());
263 for (int lev = 0; lev < repo.num_active_levels(); ++lev) {
264 fluxes[lev][0] = &(*flux_x)(lev);
265 fluxes[lev][1] = &(*flux_y)(lev);
266 fluxes[lev][2] = &(*flux_z)(lev);
267 }
268
269 // In order to enforce conservation across coarse-fine boundaries we
270 // must be sure to average down the fluxes before we use them
271 for (int lev = repo.num_active_levels() - 1; lev > 0; --lev) {
272 amrex::IntVect rr =
273 geom[lev].Domain().size() / geom[lev - 1].Domain().size();
274 amrex::average_down_faces(
275 GetArrOfConstPtrs(fluxes[lev]), fluxes[lev - 1], rr,
276 geom[lev - 1]);
277 }
278
279 for (int lev = 0; lev < repo.num_active_levels(); ++lev) {
280#ifdef AMREX_USE_OMP
281#pragma omp parallel if (amrex::Gpu::notInLaunchRegion())
282#endif
283 for (amrex::MFIter mfi(dof_field(lev), amrex::TilingIfNotGPU());
284 mfi.isValid(); ++mfi) {
285 const auto& bx = mfi.tilebox();
286
287 HydroUtils::ComputeDivergence(
288 bx, conv_term(lev).array(mfi), (*flux_x)(lev).array(mfi),
289 (*flux_y)(lev).array(mfi), (*flux_z)(lev).array(mfi),
290 PDE::ndim, geom[lev], amrex::Real(-1.0_rt),
291 fluxes_are_area_weighted);
292 }
293 }
294 }

304 godunov::scheme mflux_scheme = godunov::scheme::UPWIND;
305 std::string godunov_type{"weno_z"};
306 std::string mflux_type{"upwind"};

308 bool godunov_use_forces_in_trans{false};
309 bool m_allow_inflow_on_outflow{false};
310 std::string advection_type{"Godunov"};
311};
