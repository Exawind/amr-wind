81template <>
82struct InitDataOp<Joukowsky, ActSrcDisk>
83{
84 void operator()(typename Joukowsky::DataType& data)
85 {
86 ops::base::allocate_basic_grid_quantities<Joukowsky>(data);
87
88 auto& meta = data.meta();
89 meta.tip_correction.resize(meta.num_vel_pts_r);
90 meta.root_correction.resize(meta.num_vel_pts_r);
91 meta.f_normal.resize(meta.num_vel_pts_r);
92 meta.f_theta.resize(meta.num_force_pts);
93
94 std::fill(meta.tip_correction.begin(), meta.tip_correction.end(), 1.0);
95
96 if (meta.use_root_correction) {
97 const amrex::Real dx = 1.0 / meta.num_vel_pts_r;
98
99 const auto a = meta.root_correction_coefficient;
100 const auto b = meta.root_correction_exponent;
101
102 const auto delta = std::max(
103 meta.vortex_core_size / meta.radius(),
104 vs::DTraits<amrex::Real>::eps());
105
106 const auto factor = dx / delta;
107
108 for (int i = 0; i < meta.root_correction.size(); ++i) {
109 meta.root_correction[i] =
110 1.0 - std::exp(-a * std::pow((i + 0.5) * factor, b));
111 }
112 } else {
113 std::fill(
114 meta.root_correction.begin(), meta.root_correction.end(), 1.0);
115 }
116
117 joukowsky::update_disk_points(data);
118 }

136template <>
137struct ComputeForceOp<Joukowsky, ActSrcDisk>
138{
139 void operator()(Joukowsky::DataType& data)
140 {
141 // Equation comments refer to Sorenson 2020 (full reference above)
142 auto& grid = data.grid();
143 auto& ddata = data.meta();
144 const amrex::Real machine_eps = vs::DTraits<amrex::Real>::eps();
145 ddata.disk_force *= 0.0;
146
147 const amrex::Real uInfSqr = base::compute_reference_velocity_sqr(ddata);
148 const amrex::Real U_ref = std::max(std::sqrt(uInfSqr), machine_eps);
149 base::set_thrust_coefficient(ddata, uInfSqr);
150 joukowsky::set_current_angular_velocity(ddata, uInfSqr);
151
152 const amrex::Real Ct = ddata.current_ct;
153 const amrex::Real dx = ddata.dr / ddata.radius();
154
155 const ops::base::AreaComputer area(
156 ddata.radius(), ddata.num_vel_pts_r, ddata.num_vel_pts_t);
157
158 const amrex::Real lambda =
159 ddata.radius() * ddata.current_angular_velocity / U_ref;
160
161 ddata.current_tip_speed_ratio = lambda;
162
163 const amrex::Real lambda_2 = lambda * lambda;
164
165 // step 0: compute tip correction based on current wind speed (eq 9)
166 if (ddata.use_tip_correction) {
167 auto& f = ddata.tip_correction;
168 for (int i = 0; i < f.size(); ++i) {
169 const auto x = (i + 0.5) * dx;
170 f[i] = 2.0 / M_PI *
171 std::acos(
172 std::exp(
173 -0.5 * ddata.num_blades *
174 std::sqrt(1.0 + lambda_2) * (1.0 - x)));
175 }
176 }
177
178 // Compute the solid body term S0 (Sorensen 2022, eq. 27)
179 amrex::Real S0 = 0.0;
180 const amrex::Real alpha1 = ddata.S0_alpha1;
181 const amrex::Real alpha2 = ddata.S0_alpha2;
182
183 if (ddata.Ct_rated > 0.0) {
184 if (Ct < ddata.Ct_rated) {
185 S0 = alpha1 *
186 std::pow((ddata.Ct_rated - Ct) / ddata.Ct_rated, 3.0);
187 } else {
188 S0 = alpha2 * (ddata.Ct_rated - Ct) / ddata.Ct_rated;
189 }
190 }
191
192 // step 1: compute Ct, a1 and a2 coefficients (eq 16)
193 // a3, a4, a5 coefficients are from Sorensen, 2022
194 amrex::Real a1 = 0.0;
195 amrex::Real a2 = 0.0;
196 amrex::Real a3 = 0.0;
197 amrex::Real a4 = 0.0;
198 amrex::Real a5 = 0.0;
199
200 for (int ip = 0; ip < ddata.num_vel_pts_r; ip++) {
201 const amrex::Real x =
202 amrex::max<amrex::Real>((ip + 0.5) * dx, machine_eps);
203
204 a1 += std::pow(ddata.tip_correction[ip], 2.0) *
205 std::pow(ddata.root_correction[ip], 2.0) / x * dx;
206
207 a2 += ddata.tip_correction[ip] * ddata.root_correction[ip] * x * dx;
208
209 a3 += std::pow(ddata.tip_correction[ip], 2.0) *
210 std::pow(ddata.root_correction[ip], 2.0) * x * dx;
211
212 a4 += ddata.tip_correction[ip] * ddata.root_correction[ip] *
213 std::pow(x, 3.0) * dx;
214
215 a5 += std::pow(ddata.tip_correction[ip], 2.0) *
216 std::pow(ddata.root_correction[ip], 2.0) * std::pow(x, 3.0) *
217 dx;
218 }
219
220 // step 2: determine the circulation (q0) from a1, a2 and Ct
221 // (eq. 18, Sorensen, 2022)
222
223 // sqrt[(a2*L+a3*S0)^2+a1*(0.5*Ct-2*a4*L*S0-a5*S0^2)]-(a2*L+a3*S0)
224 // q0 = --------------------------------------------------------------
225 // a1
226
227 const amrex::Real term1 = std::pow(a2 * lambda + a3 * S0, 2.0);
228 const amrex::Real term2 =
229 a1 * (0.5 * Ct - 2.0 * a4 * lambda * S0 - a5 * S0 * S0);
230 const amrex::Real term3 = a2 * lambda + a3 * S0;
231
232 const amrex::Real numerator = std::sqrt(term1 + term2) - term3;
233
234 const amrex::Real denominator =
235 amrex::max<amrex::Real>(a1, machine_eps);
236
237 const amrex::Real q0 = numerator / denominator;
238
239 // step 3: compute normal force (fz) and azimuthal force (f_theta) (eq
240 // 13) and cp (eq 20)
241
242 VecSlice disk_velocity = ::amr_wind::utils::slice(
243 grid.vel, ddata.num_force_pts, ddata.num_force_pts);
244
245 amrex::Real moment = 0.0;
246
247 int ip = 0;
248
249 const auto dTheta = amr_wind::utils::two_pi() / ddata.num_vel_pts_t;
250 const auto current_time = data.sim().time().current_time();
251
252 for (int i = 0; i < ddata.num_vel_pts_r; i++) {
253 const amrex::Real& F = ddata.tip_correction[i];
254 const amrex::Real& g = ddata.root_correction[i];
255
256 const amrex::Real x =
257 amrex::max<amrex::Real>((i + 0.5) * dx, machine_eps);
258
259 const amrex::Real f_z =
260 (q0 / x + S0 * x) * g * F *
261 (lambda * x + 0.5 * (q0 / x + S0 * x) * g * F);
262
263 ddata.f_normal[i] = f_z;
264
265 const auto dArea = area.area_section(i);
266
267 for (int j = 0; j < ddata.num_vel_pts_t; j++, ip++) {
268 const auto point_current = grid.pos[ip];
269 const auto theta_angle = j * dTheta;
270
271 // TODO add sign convention for rotation (+/- RPM)
272 // Negative sign on theta_vec means turbine is rotating
273 // clockwise when standing upstream looking downstream
274 const auto theta_vec = -utils::compute_tangential_vector(
275 ddata.center, ddata.normal_vec, point_current);
276
277 // normal vec by definition is opposite of the wind direction
278 // so we need to flip the sign to give the actual disk velocity
279 const amrex::Real u_disk_ij =
280 -ddata.normal_vec & disk_velocity[ip];
281
282 const amrex::Real f_theta =
283 u_disk_ij / U_ref * (q0 / x + S0 * x) * g * F;
284 ddata.f_theta[ip] = f_theta;
285
286 // eq 18
287 moment += x * ddata.radius() * f_theta * dArea;
288
289 const amrex::Real f_awc =
290 ddata.awc_amplitude *
291 std::cos(
292 ddata.awc_angular_frequency * current_time -
293 ddata.awc_azimuthal_mode * theta_angle +
294 ddata.awc_clocking_angle);
295
296 grid.force[ip] = ((1 + f_awc) * f_z * ddata.normal_vec +
297 f_theta * theta_vec) *
298 ddata.density * uInfSqr * dArea;
299
300 ddata.disk_force = ddata.disk_force + grid.force[ip];
301 }
302 }
303
304 // equation 20
305 const amrex::Real eq_20_denominator = amrex::max<amrex::Real>(
306 0.5 * M_PI * ddata.density * std::pow(ddata.radius(), 2) *
307 std::pow(U_ref, 3),
308 machine_eps);
309
310 ddata.current_power =
311 ddata.density * uInfSqr * moment * ddata.current_angular_velocity;
312
313 ddata.current_cp = ddata.density * uInfSqr * moment *
314 ddata.current_angular_velocity / eq_20_denominator;
315#ifdef AMR_WIND_USE_HELICS
316 if (data.info().is_root_proc && data.sim().helics().is_activated()) {
317
318 std::cout << "turbine: " << data.info().id
319 << " jouk power: " << ddata.current_power
320 << " uinfsqr: " << uInfSqr << " moment: " << moment
321 << " ang vel: " << ddata.current_angular_velocity
322 << " normal vec: " << ddata.normal_vec[0] << ' '
323 << ddata.normal_vec[1] << std::endl;
324
325 data.sim().helics().m_turbine_power_to_controller[data.info().id] =
326 ddata.current_power;
327 const amrex::Real turbine_angle = std::atan2(
328 ddata.reference_velocity[1], ddata.reference_velocity[0]);
329 data.sim()
330 .helics()
331 .m_turbine_wind_direction_to_controller[data.info().id] =
332 270.0 - amr_wind::utils::degrees(turbine_angle);
333 }
334#endif
335 }
336};
