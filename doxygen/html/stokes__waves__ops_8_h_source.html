14template <>
15struct ReadInputsOp<StokesWaves>
16{
17 void operator()(
18 StokesWaves::DataType& data, const ::amr_wind::utils::MultiParser& pp)
19 {
20 auto& wdata = data.meta();
21 auto& info = data.info();
22 relaxation_zones::read_inputs(wdata, info, pp);
23
24 // Get gravity from MultiPhase physics, assume negative z
25 wdata.g = -data.sim().physics_manager().get<MultiPhase>().gravity()[2];
26 // Get wave attributes
27 pp.get("wave_height", wdata.wave_height);
28 pp.get("order", wdata.order);
29 if (pp.contains("wave_length")) {
30 pp.get("wave_length", wdata.wave_length);
31 } else {
32 // Wave length will be calculated using wave period
33 amrex::Real wave_period = 1.0;
34 pp.get("wave_period", wave_period);
35 // User can specify a different order for wavelength calculation,
36 // e.g., an order of 1 will use the linear dispersion relation.
37 // Default is to use the same order as the waves themselves.
38 int order_lambda = wdata.order;
39 pp.query("stokes_wavelength_order", order_lambda);
40 // Get user-specified convergence criteria, if supplied
41 amrex::Real tol_lambda = 1e-10;
42 int itmax_lambda = 40;
43 pp.query("stokes_wavelength_tolerance", tol_lambda);
44 pp.query("stokes_wavelength_iter_max", itmax_lambda);
45 wdata.wave_length = relaxation_zones::stokes_wave_length(
46 wave_period, wdata.water_depth, wdata.wave_height, order_lambda,
47 wdata.g, tol_lambda, itmax_lambda);
48 // Abort if wave length is negative
49 if (wdata.wave_length < 0.0) {
50 if (wdata.wave_length == -1) {
51 amrex::Print() << "Stokes wave length too close to 0.\n";
52 }
53 if (wdata.wave_length == -2) {
54 amrex::Print() << "Stokes wave length is not a number.\n";
55 }
56 if (wdata.wave_length == -3) {
57 amrex::Print() << "Stokes wave length calculation used "
58 "maximum iterations before converging.\n";
59 }
60 amrex::Abort(
61 "Failed to properly calculate a wave length in "
62 "stokes_waves_ops.H");
63 } else {
64 amrex::Print() << "Stokes wave length calculated to be "
65 << wdata.wave_length << " m.\n";
66 }
67 }
68 }
