111 c = Stencil::c01;
112 cm1 = Stencil::c02;
113 const amrex::Real phix =
114 (cp1 * phi(i + 1, j, k, n) + c * phi(i, j, k, n) +
115 cm1 * phi(i - 1, j, k, n)) *
116 idx[0];

127 (phiz * phiz * phiyy)) /
128 std::pow(
129 (phix * phix) + (phiy * phiy) + (phiz * phiz), 1.5_rt);
