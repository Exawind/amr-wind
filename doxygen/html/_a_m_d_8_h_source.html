63AMREX_GPU_DEVICE AMREX_FORCE_INLINE amrex::Real amd_muvel(
64 int i,
65 int j,
66 int k,
67 const amrex::GpuArray<amrex::Real, AMREX_SPACEDIM>& dx, // Grid spacing
68 const amrex::Real beta, // Thermal expansion coefficient
69 const amrex::GpuArray<amrex::Real, AMREX_SPACEDIM>& gravity,
70 const amrex::Real C, // Poincare const
71 const amrex::Array4<amrex::Real const>& gradVel,
72 const amrex::Array4<amrex::Real const>& gradT,
73 const amrex::Real* gradTbar_coord_begin,
74 const amrex::Real* gradTbar_coord_end,
75 const amrex::Real* gradTbar,
76 const int normal_dir,
77 const amrex::Real nlo) noexcept
78{
79 amrex::Real num_shear = 0;
80 amrex::Real num_buoy = 0;
81 amrex::Real denom = 0;
82 const int n_ind = amr_wind::direction_selector(i, j, k, normal_dir);
83 const amrex::Real h = nlo + (n_ind + 0.5) * dx[normal_dir];
84 const amrex::Real gradTbar_h = amr_wind::interp::linear(
85 gradTbar_coord_begin, gradTbar_coord_end, gradTbar, h);
86 for (int ii = 0; ii < AMREX_SPACEDIM; ++ii) {
87 // This should operate only on the wall normal velocity
88 num_buoy +=
89 gradVel(i, j, k, normal_dir * AMREX_SPACEDIM + ii) *
90 (gradT(i, j, k, ii) - ((ii == normal_dir) ? gradTbar_h : 0.0)) *
91 dx[ii] * dx[ii];
92
93 for (int jj = 0; jj < AMREX_SPACEDIM; ++jj) {
94 const amrex::Real diuj = gradVel(i, j, k, ii * AMREX_SPACEDIM + jj);
95 const amrex::Real djui = gradVel(i, j, k, jj * AMREX_SPACEDIM + ii);
96 denom += diuj * diuj;
97 const amrex::Real sij = 0.5 * (diuj + djui);
98 for (int kk = 0; kk < AMREX_SPACEDIM; ++kk) {
99 const amrex::Real dkui =
100 gradVel(i, j, k, ii * AMREX_SPACEDIM + kk);
101 const amrex::Real dkuj =
102 gradVel(i, j, k, jj * AMREX_SPACEDIM + kk);
103 num_shear += dkui * dkuj * dx[kk] * dx[kk] * sij;
104 }
105 }
106 }
107 denom = std::max(1e-15, denom);
108 num_shear *= -C;
109 num_buoy *= C * beta * std::abs(gravity[normal_dir]);
110 return std::max(1e-15, (num_shear + num_buoy) / denom);
111}

113AMREX_GPU_DEVICE AMREX_FORCE_INLINE amrex::Real amd_thermal_diff(
114 int i,
115 int j,
116 int k,
117 const amrex::GpuArray<amrex::Real, AMREX_SPACEDIM>& dx, // Grid spacing
118 const amrex::Real C, // Poincare const
119 const amrex::Array4<amrex::Real const>& gradVel,
120 const amrex::Array4<amrex::Real const>& gradT) noexcept
121{
122 amrex::Real num = 0;
123 amrex::Real denom = 0;
124 for (int ii = 0; ii < AMREX_SPACEDIM; ++ii) {
125 const amrex::Real diT = gradT(i, j, k, ii);
126 denom += diT * diT;
127 for (int kk = 0; kk < AMREX_SPACEDIM; ++kk) {
128 const amrex::Real dkui = gradVel(i, j, k, ii * AMREX_SPACEDIM + kk);
129 const amrex::Real dkT = gradT(i, j, k, kk);
130 num += dkui * diT * dkT * dx[kk] * dx[kk];
131 }
132 }
133
134 denom = std::max(1e-15, denom);
135 num *= -C;
136 return std::max(1e-15, num / denom);
137}
