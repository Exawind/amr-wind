66AMREX_GPU_DEVICE AMREX_FORCE_INLINE amrex::Real amd_muvel(
67 int i,
68 int j,
69 int k,
70 const amrex::GpuArray<amrex::Real, AMREX_SPACEDIM>& dx, // Grid spacing
71 const amrex::Real beta, // Thermal expansion coefficient
72 const amrex::GpuArray<amrex::Real, AMREX_SPACEDIM>& gravity,
73 const amrex::Real C, // Poincare const
74 const amrex::Array4<amrex::Real const>& gradVel,
75 const amrex::Array4<amrex::Real const>& gradT,
76 const amrex::Real* gradTbar_coord_begin,
77 const amrex::Real* gradTbar_coord_end,
78 const amrex::Real* gradTbar,
79 const int normal_dir,
80 const amrex::Real nlo)
81{
82 amrex::Real num_shear = 0.0_rt;
83 amrex::Real num_buoy = 0.0_rt;
84 amrex::Real denom = 0.0_rt;
85 const int n_ind = amr_wind::direction_selector(i, j, k, normal_dir);
86 const amrex::Real h = nlo + ((n_ind + 0.5_rt) * dx[normal_dir]);
87 const amrex::Real gradTbar_h = amr_wind::interp::linear(
88 gradTbar_coord_begin, gradTbar_coord_end, gradTbar, h);
89 for (int ii = 0; ii < AMREX_SPACEDIM; ++ii) {
90 // This should operate only on the wall normal velocity
91 num_buoy +=
92 gradVel(i, j, k, (normal_dir * AMREX_SPACEDIM) + ii) *
93 (gradT(i, j, k, ii) - ((ii == normal_dir) ? gradTbar_h : 0.0_rt)) *
94 dx[ii] * dx[ii];
95
96 for (int jj = 0; jj < AMREX_SPACEDIM; ++jj) {
97 const amrex::Real diuj =
98 gradVel(i, j, k, (ii * AMREX_SPACEDIM) + jj);
99 const amrex::Real djui =
100 gradVel(i, j, k, (jj * AMREX_SPACEDIM) + ii);
101 denom += diuj * diuj;
102 const amrex::Real sij = 0.5_rt * (diuj + djui);
103 for (int kk = 0; kk < AMREX_SPACEDIM; ++kk) {
104 const amrex::Real dkui =
105 gradVel(i, j, k, (ii * AMREX_SPACEDIM) + kk);
106 const amrex::Real dkuj =
107 gradVel(i, j, k, (jj * AMREX_SPACEDIM) + kk);
108 num_shear += dkui * dkuj * dx[kk] * dx[kk] * sij;
109 }
110 }
111 }
112 denom = amrex::max<amrex::Real>(
113 std::numeric_limits<amrex::Real>::epsilon() * 1.0e1_rt, denom);
114 num_shear *= -C;
115 num_buoy *= C * beta * std::abs(gravity[normal_dir]);
116 return amrex::max<amrex::Real>(
117 std::numeric_limits<amrex::Real>::epsilon() * 1.0e1_rt,
118 (num_shear + num_buoy) / denom);
119}

121AMREX_GPU_DEVICE AMREX_FORCE_INLINE amrex::Real amd_thermal_diff(
122 int i,
123 int j,
124 int k,
125 const amrex::GpuArray<amrex::Real, AMREX_SPACEDIM>& dx, // Grid spacing
126 const amrex::Real C, // Poincare const
127 const amrex::Array4<amrex::Real const>& gradVel,
128 const amrex::Array4<amrex::Real const>& gradT)
129{
130 amrex::Real num = 0;
131 amrex::Real denom = 0;
132 for (int ii = 0; ii < AMREX_SPACEDIM; ++ii) {
133 const amrex::Real diT = gradT(i, j, k, ii);
134 denom += diT * diT;
135 for (int kk = 0; kk < AMREX_SPACEDIM; ++kk) {
136 const amrex::Real dkui =
137 gradVel(i, j, k, (ii * AMREX_SPACEDIM) + kk);
138 const amrex::Real dkT = gradT(i, j, k, kk);
139 num += dkui * diT * dkT * dx[kk] * dx[kk];
140 }
141 }
142
143 denom = amrex::max<amrex::Real>(
144 std::numeric_limits<amrex::Real>::epsilon() * 1.0e1_rt, denom);
145 num *= -C;
146 return amrex::max<amrex::Real>(
147 std::numeric_limits<amrex::Real>::epsilon() * 1.0e1_rt, num / denom);
148}
