145 void compute_diff_term(const FieldState fstate)
146 {
147 this->setup_operator(*this->m_applier, 0.0, -1.0, fstate);
148
149 auto tau_state = std::is_same<Scheme, fvm::Godunov>::value
150 ? FieldState::New
151 : fstate;
152
153 amrex::MLMG mlmg(*this->m_applier);
154 mlmg.apply(
155 this->m_pdefields.diff_term.state(tau_state).vec_ptrs(),
156 this->m_pdefields.field.vec_ptrs());
157
158 const auto& repo = this->m_pdefields.repo;
159 const int nlevels = repo.num_active_levels();
160 auto& divtau = this->m_pdefields.diff_term.state(tau_state);
161 const auto& density =
162 this->m_pdefields.repo.get_field("density").state(FieldState::NPH);
163 if (!PDE::multiply_rho) {
164 for (int lev = 0; lev < nlevels; ++lev) {
165 const auto& divtau_arrs = divtau(lev).arrays();
166 const auto& rho_arrs = density(lev).const_arrays();
167
168 amrex::ParallelFor(
169 divtau(lev), divtau.num_grow(), divtau.num_comp(),
170 [=] AMREX_GPU_DEVICE(
171 int nbx, int i, int j, int k, int n) noexcept {
172 const amrex::Real rhoinv = 1.0 / rho_arrs[nbx](i, j, k);
173 divtau_arrs[nbx](i, j, k, n) *= rhoinv;
174 });
175 }
176 amrex::Gpu::streamSynchronize();
177 }
178 }
