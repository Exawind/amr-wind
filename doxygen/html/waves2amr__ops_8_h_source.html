23 const int new_ntime,

25 const amrex::Real wdt,
26 const amrex::Real time)

28 // Flag to indicate that data must be read twice for interpolation
29 int double_data = 0;
30 // Check if time indicates reading must take place
31 if (new_ntime != ntime) {
32 // New data is needed - reading should happen
33 read_flag = true;
34 // If time index has proceeded more than 1 step
35 if (new_ntime > ntime + 1) {
36 // Double reading is necessary
37 double_data = 1;
38 }
39 // Time index for reading
40 ntime = new_ntime;
41 // Sim time to go with recorded data
42 wtime = new_ntime * wdt - wtinit;
43 // If double reading is deemed necessary, check for convenience
44 if (double_data == 1 && std::abs(wtime - time) <= 1e-10) {
45 // Reading can be done just once, w2a fields replace ow fields
46 double_data = 2;
47 }
48 }
49 // Check if reading must take place for other reasons
50 if (t_last < -1e-10) {
51 // Signifies initialization from scratch without waves or a restart
52 read_flag = true;
53 // Resizing needs to happen for the first time
54 resize_flag = true;
55
56 // Confirm that new time is not coincident with modes time step
57 if (std::abs(wtime - time) > 1e-10) {
58 // Data must be read before and after wtime
59 double_data = 1;
60 } else {
61 // Data need only be read once, w2a fields will replace ow fields
62 double_data = 2;
63 }
64
65 } else if (std::abs(t_last) < 1e-10) {
66 // Signifies initialization with waves
67 read_flag = true;
68 // Resizing needs to happen for the first time
69 resize_flag = true;
70
71 // levelset and velocity fields are up-to-date at t=0
72 // interpolation is ready to go
73 }
74 // Record latest time as 'last' for next timestep
75 t_last = time;
76 // Return flag regarding double reading
77 return double_data;
78}

177void time_interpolate_wave_fields(
178 amr_wind::Field& lvs_field,
179 amr_wind::Field& vel_field,
180 const amr_wind::Field& future_lvs_field,
181 const amr_wind::Field& future_vel_field,
182 const amrex::Real t_old,
183 const amrex::Real t_new,
184 const amrex::Real t_future)
185{
186 const amrex::Real b = (t_new - t_old) / (t_future - t_old + 1e-16);
187 const amrex::Real a = 1. - b;
188 amr_wind::field_ops::lincomb(
189 lvs_field, a, lvs_field, 0, b, future_lvs_field, 0, 0, 1,
190 lvs_field.num_grow());
191 amr_wind::field_ops::lincomb(
192 vel_field, a, vel_field, 0, b, future_vel_field, 0, 0, AMREX_SPACEDIM,
193 vel_field.num_grow());
194}

586template <>
587struct InitDataOp<W2AWaves>
588{
589 void
590 // cppcheck-suppress constParameterReference
591 operator()(
592 W2AWaves::DataType& data,
593 int level,
594 const amrex::Geometry& geom,
595 bool multiphase_mode)
596 {
597
598#ifdef AMR_WIND_USE_W2A
599 auto& wdata = data.meta();
600
601 auto& sim = data.sim();
602
603 // Fill ow fields, then populate flow fields according to setup
604 auto& ow_levelset = sim.repo().get_field("ow_levelset");
605 auto& ow_velocity = sim.repo().get_field("ow_velocity");
606
607 auto& velocity = sim.repo().get_field("velocity");
608 // Set w2a fields to default values to prep for updates
609 auto& w2a_levelset = sim.repo().get_field("w2a_levelset");
610 auto& w2a_velocity = sim.repo().get_field("w2a_velocity");
611
612 Field* levelset{nullptr};
613 if (multiphase_mode) {
614 levelset = &sim.repo().get_field("levelset");
615 }
616
617 const auto& problo = geom.ProbLoArray();
618 const auto& probhi = geom.ProbHiArray();
619 const auto& dx = geom.CellSizeArray();
620
621 // Set t_last to 0.0 to signify information read in
622 wdata.t_last = 0.0;
623
624 // indvec is complete upon initialization (all heights every proc)
625 amrex::Vector<int> indvec(wdata.hvec.size());
626 std::iota(indvec.begin(), indvec.end(), 0);
627 // Interpolate to MultiFabs (one level at a time)
628 interp_to_mfab::interp_eta_to_levelset_multifab(
629 wdata.n0_sp, wdata.n1_sp, wdata.dx0, wdata.dx1, wdata.xlo0,
630 wdata.xlo1, wdata.zsl, wdata.is_ocean, wdata.sp_eta_vec,
631 ow_levelset(level), problo, dx);
632 interp_to_mfab::interp_velocity_to_multifab(
633 wdata.n0_sp, wdata.n1_sp, wdata.dx0, wdata.dx1, wdata.xlo0,
634 wdata.xlo1, wdata.is_ocean, indvec, wdata.hvec, wdata.sp_u_vec,
635 wdata.sp_v_vec, wdata.sp_w_vec, ow_velocity(level), problo, dx);
636
637 // Populate flow fields according to intended forcing and init setup
638 const auto& ow_phi = ow_levelset(level).const_arrays();
639 const auto& ow_vel = ow_velocity(level).const_arrays();
640 const auto& vel = velocity(level).arrays();
641 const auto& phi_arrs = multiphase_mode
642 ? (*levelset)(level).arrays()
643 : amrex::MultiArray4<amrex::Real>();
644
645 const auto& w2a_phi = w2a_levelset(level).arrays();
646 const auto& w2a_vel = w2a_velocity(level).arrays();
647
648 const amrex::Real gen_length = wdata.gen_length;
649 const amrex::Real beach_length = wdata.beach_length;
650 const amrex::Real zero_sea_level = wdata.zsl;
651
652 const bool has_beach = wdata.has_beach && multiphase_mode;
653 const bool init_wave_field = wdata.init_wave_field || !multiphase_mode;
654
655 amrex::MultiFab phi_tmp_fab(
656 velocity(level).boxArray(), velocity(level).DistributionMap(), 1,
657 3);
658 amrex::MultiFab vel_tmp_fab(
659 velocity(level).boxArray(), velocity(level).DistributionMap(),
660 AMREX_SPACEDIM, 3);
661
662 const auto& phi_tmp = phi_tmp_fab.arrays();
663 const auto& vel_tmp = vel_tmp_fab.arrays();
664
665 amrex::ParallelFor(
666 velocity(level), amrex::IntVect(3),
667 [=] AMREX_GPU_DEVICE(int nbx, int i, int j, int k) noexcept {
668 const amrex::Real x = problo[0] + (i + 0.5) * dx[0];
669 const amrex::Real z = problo[2] + (k + 0.5) * dx[2];
670
671 // Wave profile
672 const utils::WaveVec wave_sol{
673 ow_vel[nbx](i, j, k, 0), ow_vel[nbx](i, j, k, 1),
674 ow_vel[nbx](i, j, k, 2), ow_phi[nbx](i, j, k) + z};
675 // Quiescent profile
676 const utils::WaveVec quiescent{0.0, 0.0, 0.0, zero_sea_level};
677
678 // Specify initial state for each region of domain
679 const auto bulk = init_wave_field ? wave_sol : quiescent;
680 const auto outlet = has_beach ? quiescent : wave_sol;
681
682 const auto local_profile = utils::harmonize_profiles_1d(
683 x, problo[0], gen_length, probhi[0], beach_length, wave_sol,
684 bulk, outlet);
685
686 phi_tmp[nbx](i, j, k) = local_profile[3] - z;
687 vel_tmp[nbx](i, j, k, 0) = local_profile[0];
688 vel_tmp[nbx](i, j, k, 1) = local_profile[1];
689 vel_tmp[nbx](i, j, k, 2) = local_profile[2];
690
691 if (multiphase_mode) {
692 phi_arrs[nbx](i, j, k) = phi_tmp[nbx](i, j, k);
693 }
694
695 // Default w2a values matter for where no updates happen
696 w2a_phi[nbx](i, j, k) = quiescent[3] - z;
697 w2a_vel[nbx](i, j, k, 0) = quiescent[0];
698 w2a_vel[nbx](i, j, k, 1) = quiescent[0];
699 w2a_vel[nbx](i, j, k, 2) = quiescent[0];
700 });
701 amrex::Gpu::streamSynchronize();
702
703 amrex::ParallelFor(
704 velocity(level), amrex::IntVect(2),
705 [=] AMREX_GPU_DEVICE(int nbx, int i, int j, int k) noexcept {
706 const amrex::Real eps = 2. * std::cbrt(dx[0] * dx[1] * dx[2]);
707 const amrex::Real vof_local =
708 multiphase::levelset_to_vof(i, j, k, eps, phi_tmp[nbx]);
709
710 if (vof_local > 1. - constants::TIGHT_TOL) {
711 // Entire cell is liquid
712 vel[nbx](i, j, k, 0) = vel_tmp[nbx](i, j, k, 0);
713 vel[nbx](i, j, k, 1) = vel_tmp[nbx](i, j, k, 1);
714 vel[nbx](i, j, k, 2) = vel_tmp[nbx](i, j, k, 2);
715 } else if (vof_local > 1e-3) {
716 // Velocity of liquid becomes velocity of entire cell
717 // Interpolate using cell-centered values to liquid centroid
718 // Consider only z-direction to find centroid
719 const amrex::Real z_cell = problo[2] + (k + 0.5) * dx[2];
720 const amrex::Real z_cell_below = z_cell - dx[2];
721 const amrex::Real z_cell_bottom = z_cell - 0.5 * dx[2];
722 const amrex::Real z_liq_center =
723 z_cell_bottom + 0.5 * vof_local * dx[2];
724 for (int n = 0; n < AMREX_SPACEDIM; ++n) {
725 vel[nbx](i, j, k, n) =
726 vel_tmp[nbx](i, j, k - 1, n) +
727 (vel_tmp[nbx](i, j, k, n) -
728 vel_tmp[nbx](i, j, k - 1, n)) *
729 ((z_liq_center - z_cell_below) /
730 (z_cell - z_cell_below + constants::EPS));
731 }
732 }
733 });
734 amrex::Gpu::streamSynchronize();
735#else
736 amrex::ignore_unused(data, level, geom, multiphase_mode);
737#endif
738 }
739}; // namespace ocean_waves

741template <>
742struct UpdateTargetFieldsOp<W2AWaves>
743{
744 // cppcheck-suppress constParameterReference
745 void operator()(W2AWaves::DataType& data, const amrex::Real time)
746 {
747
748#ifdef AMR_WIND_USE_W2A
749 auto& wdata = data.meta();
750 auto& sim = data.sim();
751
752 // Update ow fields every time
753 auto& ow_levelset = sim.repo().get_field("ow_levelset");
754 auto& ow_velocity = sim.repo().get_field("ow_velocity");
755 // Update HOS fields when necessary
756 auto& w2a_levelset = sim.repo().get_field("w2a_levelset");
757 auto& w2a_velocity = sim.repo().get_field("w2a_velocity");
758
759 // Proxy for multiphase mode of ocean waves
760 bool vof_exists = sim.repo().field_exists("vof");
761
762 auto nlevels = sim.repo().num_active_levels();
763 auto geom = sim.mesh().Geom();
764
765 // Get value for time interpolation
766 amrex::Real t_last = wdata.t_last;
767
768 // Check if new HOS data needs to be read
769 bool read_flag = false;
770 // Check if time indicates reading must take place
771 int new_ntime =
772 wdata.is_ocean
773 ? wdata.c_rmodes.time2step(time + wdata.t_winit, wdata.ntime)
774 : wdata.r_rmodes.time2step(time + wdata.t_winit, wdata.ntime);
775 int double_data = evaluate_read_resize(
776 wdata.ntime, read_flag, wdata.resize_flag, wdata.t, wdata.t_last,
777 new_ntime, wdata.t_winit, wdata.dt_modes, time);
778 // Check if stop time is exceeded, introduce offset to ntime
779 if (read_flag) {
780 // Need to only check when reading is occurring
781 if (wdata.ntime + wdata.n_offset > wdata.n_wstop) {
782 // If exceeding stop step, calculate new offset
783 wdata.n_offset =
784 update_offset_timestep(wdata.ntime, wdata.n_winit);
785 // Print warning to screen
786 amrex::Print()
787 << "WARNING (waves2amr_ops): available mode data exceeded, "
788 "resetting to beginning of mode data.\n";
789 }
790 }
791 // Resizing (assuming reading is taking place) must happen after regrid
792 if (wdata.regrid_occurred) {
793 // resize_flag remains true until resizing occurs, but
794 // regrid_occurred resets every timestep
795 wdata.resize_flag = true;
796 }
797
798 // Read HOS data if necessary based on time
799 if (read_flag) {
800
801 amrex::Print() << "OceanWaves: Waves2AMR reading from file: step "
802 << wdata.ntime << std::endl
803 << " last interp time " << t_last << std::endl
804 << " simulation time " << time << std::endl
805 << " new data time " << wdata.t << std::endl
806 << " double data read " << double_data
807 << std::endl;
808
809 if (wdata.resize_flag) {
810 // Reset flag
811 wdata.resize_flag = false;
812 // Flags for indicating overlap, assume none at first
813 bool flag_z = false;
814 bool flag_xlo = false;
815 bool flag_xhi = false;
816 bool flag_ylo = false;
817 bool flag_yhi = false;
818 // Get heights for this processor, check overlap in z
819 flag_z =
820 (interp_to_mfab::get_local_height_indices(
821 wdata.indvec, wdata.hvec, ow_velocity.vec_ptrs(),
822 geom) == 1);
823 // No overlap from heights definitely means no interp
824
825 // Check lateral bounds (in x)
826 flag_xlo =
827 (interp_to_mfab::check_lateral_overlap_lo(
828 wdata.gen_length, 0, ow_velocity.vec_ptrs(), geom) ==
829 1);
830 // No overlap with gen region means no interp, unless ...
831 if (!wdata.has_beach) {
832 // ... if overlap exists here, needing interp
833 flag_xhi =
834 (interp_to_mfab::check_lateral_overlap_hi(
835 wdata.beach_length, 0, ow_velocity.vec_ptrs(),
836 geom) == 1);
837 }
838 // Then in y, if y zones are present
839 if (wdata.zone_length_y > constants::EPS) {
840 flag_ylo =
841 (interp_to_mfab::check_lateral_overlap_lo(
842 wdata.zone_length_y, 1, ow_velocity.vec_ptrs(),
843 geom) == 1);
844 flag_yhi =
845 (interp_to_mfab::check_lateral_overlap_hi(
846 wdata.zone_length_y, 1, ow_velocity.vec_ptrs(),
847 geom) == 1);
848 }
849
850 // Wave generation zones are irrelevant for single-phase mode,
851 // indicated by nonexistence of vof
852 if (!vof_exists) {
853 flag_xlo = true;
854 flag_xhi = true;
855 }
856
857 if (flag_z && (flag_xlo || flag_xhi || flag_ylo || flag_yhi)) {
858 // Interpolation is needed
859 wdata.do_interp = true;
860 // Do resizing
861 wdata.sp_eta_vec.resize(
862 static_cast<size_t>(wdata.n0) *
863 static_cast<size_t>(wdata.n1),
864 0.0);
865 wdata.sp_u_vec.resize(
866 static_cast<size_t>(wdata.n0 * wdata.n1) *
867 wdata.indvec.size());
868 wdata.sp_v_vec.resize(
869 static_cast<size_t>(wdata.n0 * wdata.n1) *
870 wdata.indvec.size());
871 wdata.sp_w_vec.resize(
872 static_cast<size_t>(wdata.n0 * wdata.n1) *
873 wdata.indvec.size());
874 // Sizes will remain constant and need for interpolation
875 // will remain until a regrid occurs
876 } else {
877 // No overlapping with spatial data or no overlapping with
878 // relaxation zones, interpolation can be skipped
879 wdata.do_interp = false;
880 }
881 }
882 // Only perform reading where needed, communicate offset though
883 amrex::ParallelDescriptor::ReduceIntMax(wdata.n_offset);
884
885 // Previous W2A data is more recent than last interpolated data
886 t_last = (wdata.ntime - 1) * wdata.dt_modes;
887
888 if (double_data == 0) {
889 // Previous W2A data needs to be copied into ow fields
890 amr_wind::field_ops::copy(
891 ow_levelset, w2a_levelset, 0, 0, 1, ow_levelset.num_grow());
892 amr_wind::field_ops::copy(
893 ow_velocity, w2a_velocity, 0, 0, AMREX_SPACEDIM,
894 ow_velocity.num_grow());
895 } else if (double_data == 1) {
896 // Previous W2A data is unknown, read into ow fields
897 if (wdata.do_interp) {
898 populate_fields_all_levels(
899 wdata, geom, ow_levelset, ow_velocity, -1);
900 }
901
902 // Average down to get fine information on coarse grid where
903 // possible (may be unnecessary)
904 for (int lev = nlevels - 1; lev > 0; --lev) {
905 amrex::average_down(
906 ow_velocity(lev), ow_velocity(lev - 1), 0,
907 AMREX_SPACEDIM, sim.mesh().refRatio(lev - 1));
908 amrex::average_down(
909 ow_levelset(lev), ow_levelset(lev - 1), 0, 1,
910 sim.mesh().refRatio(lev - 1));
911 }
912 // Fill patch to get correct ghost cells after average down
913 ow_velocity.fillpatch(sim.time().new_time());
914 ow_levelset.fillpatch(sim.time().new_time());
915 }
916 // ow fields cancel when double_data == 2, no modification
917
918 // After possible prior read, now read data for this ntime
919 if (wdata.do_interp) {
920 populate_fields_all_levels(
921 wdata, geom, w2a_levelset, w2a_velocity);
922 }
923
924 // Average down to get fine information on coarse grid where
925 // possible and update ghost cells (may be unnecessary)
926 for (int lev = nlevels - 1; lev > 0; --lev) {
927 amrex::average_down(
928 w2a_velocity(lev), w2a_velocity(lev - 1), 0, AMREX_SPACEDIM,
929 sim.mesh().refRatio(lev - 1));
930 w2a_velocity(lev - 1).FillBoundary(geom[lev - 1].periodicity());
931 amrex::average_down(
932 w2a_levelset(lev), w2a_levelset(lev - 1), 0, 1,
933 sim.mesh().refRatio(lev - 1));
934 w2a_levelset(lev - 1).FillBoundary(geom[lev - 1].periodicity());
935 }
936 }
937
938 // Temporally interpolate at every timestep to get target solution
939 time_interpolate_wave_fields(
940 ow_levelset, ow_velocity, w2a_levelset, w2a_velocity, t_last, time,
941 wdata.t);
942#else
943 amrex::ignore_unused(data, time);
944#endif
945 }
946};
